<script>
let lines = [
  "Civilization did not begin with us...",
  "But it will be shaped by us.",
  "In this world, every decision matters.",
  "Welcome to STEM Nation."
];

function startIntro() {
  let i = 0;
  let el = document.getElementById("text");

  let interval = setInterval(() => {
    el.innerText = lines[i];
    i++;
    if(i >= lines.length) {
      clearInterval(interval);
      setTimeout(() => {
        window.location.href = "https://stem-nation.onrender.com/";
      }, 2000);
    }
  }, 2500);
}
</script>
